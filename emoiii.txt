-- --- Tab Buttons ---
local tabs = {}

local function createTab(name)
    local TabButton = Instance.new("TextButton")
    TabButton.Name = name .. "Tab"
    TabButton.Size = UDim2.new(0, 130, 0, 40)
    TabButton.BackgroundColor3 = Color3.fromRGB(20, 20, 20)
    TabButton.Text = "  " .. name -- Simple padding
    TabButton.TextColor3 = Color3.fromRGB(200, 200, 200)
    TabButton.Font = Enum.Font.GothamSemibold
    TabButton.TextSize = 14
    TabButton.TextXAlignment = Enum.TextXAlignment.Left
    TabButton.Parent = Sidebar

    local Corner = Instance.new("UICorner")
    Corner.CornerRadius = UDim.new(0, 6)
    Corner.Parent = TabButton

    local Stroke = Instance.new("UIStroke")
    Stroke.Color = Color3.fromRGB(0, 170, 255)
    Stroke.Thickness = 0 -- Off by default
    Stroke.Parent = TabButton

    -- Placeholder for the content container
    local ContentContainer = Instance.new("ScrollingFrame")
    ContentContainer.Name = name .. "Content"
    ContentContainer.Size = UDim2.new(1, -160, 1, -60)
    ContentContainer.Position = UDim2.new(0, 160, 0, 60)
    ContentContainer.BackgroundTransparency = 1
    ContentContainer.CanvasSize = UDim2.new(0, 0, 0, 0) -- Automatically calculated
    ContentContainer.ScrollBarThickness = 3
    ContentContainer.ScrollBarImageColor3 = Color3.fromRGB(0, 170, 255)
    ContentContainer.Visible = false
    ContentContainer.Parent = MainFrame

    local ContentLayout = Instance.new("UIListLayout")
    ContentLayout.Padding = UDim.new(0, 10)
    ContentLayout.HorizontalAlignment = Enum.HorizontalAlignment.Center
    ContentLayout.Parent = ContentContainer

    -- Handle Content Auto-Scrolling
    ContentLayout:GetPropertyChangedSignal("AbsoluteContentSize"):Connect(function()
        ContentContainer.CanvasSize = UDim2.new(0, 0, 0, ContentLayout.AbsoluteContentSize.Y)
    end)

    tabs[name] = {Button = TabButton, Container = ContentContainer, Stroke = Stroke}

    TabButton.MouseButton1Click:Connect(function()
        if activeTab == name then return end

        -- Hide previous tab
        if activeTab then
            tabs[activeTab].Container.Visible = false
            tabs[activeTab].Stroke.Thickness = 0
            tabs[activeTab].Button.TextColor3 = Color3.fromRGB(200, 200, 200)
            tabs[activeTab].Button.BackgroundColor3 = Color3.fromRGB(20, 20, 20)
        end

        -- Show new tab
        ContentContainer.Visible = true
        Stroke.Thickness = 1.5 -- Cyan border for active tab
        TabButton.TextColor3 = Color3.fromRGB(255, 255, 255)
        TabButton.BackgroundColor3 = Color3.fromRGB(30, 30, 30)
        activeTab = name
    end)
end

-- B. FUNCTION: ESP QUÁI (VISUALS)
-- Extracted logic from Nakhubv2.txt lines 30-49 (Black Circle ESP)
_G.MobESPVisible = false
createFunctionButton("👁️ Mob ESP (Circles)", "Render black circle ESP on mobs (Visuals)", function()
    _G.MobESPVisible = not _G.MobESPVisible
    game:GetService("StarterGui"):SetCore("SendNotification", {Title = "Linh Hub", Text = "Toggle ESP: " .. tostring(_G.MobESPVisible), Duration = 3})
    
    local circles = {}
    local function createCircle()
        local circle = Drawing.new("Circle")
        circle.Color = Color3.fromRGB(0, 0, 0) -- Pure Black per NakHub
        circle.Thickness = 2
        circle.Radius = 1.2
        circle.Visible = false
        return circle
    end

    RunService.RenderStepped:Connect(function()
        if not _G.MobESPVisible then
            for _, circle in pairs(circles) do circle.Visible = false end
            return
        end

        for _, v in pairs(workspace.Enemies:GetChildren()) do
            if v:FindFirstChild("HumanoidRootPart") then
                local pos, onScreen = Camera:WorldToViewportPoint(v.HumanoidRootPart.Position)
                if onScreen then
                    local circle = circles[v] or createCircle()
                    circles[v] = circle
                    circle.Position = Vector2.new(pos.X, pos.Y)
                    circle.Visible = true
                elseif circles[v] then
                    circles[v].Visible = false
                end
            end
        end
    end)
end)

-- C. FUNCTION: FPS RAINBOW COUNTER (VISUALS/MISC)
-- Extracted logic from Nakhubv2.txt lines 51-64
local fpsCounterOn = false
createFunctionButton("📊 FPS Counter (Rainbow)", "Display FPS with shifting colors", function()
    if fpsCounterOn then return end
    fpsCounterOn = true
    game:GetService("StarterGui"):SetCore("SendNotification", {Title = "Linh Hub", Text = "Kích hoạt Rainbow FPS Counter", Duration = 3})

    local label = Instance.new("TextLabel", ScreenGui)
    label.Size = UDim2.new(0, 100, 0, 30)
    label.Position = UDim2.new(0, 10, 0, 10)
    label.BackgroundTransparency = 1
    label.TextColor3 = Color3.new(1,1,1)
    label.TextSize = 20
    
    local frameCount = 0
    local lastUpdate = tick()
    
    spawn(function()
        while fpsCounterOn do
            frameCount = frameCount + 1
            if tick() - lastUpdate >= 1 then
                label.Text = "FPS: " .. math.floor(frameCount / (tick() - lastUpdate))
                frameCount = 0
                lastUpdate = tick()
            end
            -- Đổi màu rainbow
            label.TextColor3 = Color3.fromHSV(tick() % 5 / 5, 1, 1)
            task.wait()
        end
    end)
end)
